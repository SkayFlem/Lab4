int main() {
    int A, B, C;

    printf("Введите три числа через пробел: ");
    scanf("%d %d %d", &A, &B, &C);

    if (A % 3 == 0 && B % 3 == 0 && C % 3 == 0) {
        printf("Гипотеза подтверждена: все числа кратны трем.\n");
    } else {
        printf("Гипотеза не подтверждена: не все числа кратны трем.\n");

    return 0;
}
